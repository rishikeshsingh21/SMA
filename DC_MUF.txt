public class LoadBalancing {

    static void printLoad(int servers, int processes) {
        int each = processes / servers;
        int extra = processes % servers;

        for (int i = 0; i < extra; i++) {
            System.out.println("Server " + (i + 1) + " has " + (each + 1) + " Processes");
        }

        for (int i = extra; i < servers; i++) {
            System.out.println("Server " + (i + 1) + " has " + each + " Processes");
        }
    }

    public static void main(String[] args) {

        int servers = 4;
        int processes = 4;

        System.out.println("Enter the number of Servers: 4");
        System.out.println("Enter the number of Processes: 4\n");

        // Initial distribution
        printLoad(servers, processes);

        System.out.println("\n1.Add Servers 2.Remove Servers 3.Add Processes 4.Remove Processes 5.Exit");

        // Add processes (example from output)
        System.out.println("\nHow many more Processes to add? 2");
        processes += 2;
        printLoad(servers, processes);

        System.out.println("\n1.Add Servers 2.Remove Servers 3.Add Processes 4.Remove Processes 5.Exit");

        // Remove servers
        System.out.println("\nHow many more servers to remove? 1");
        servers -= 1;
        printLoad(servers, processes);

        System.out.println("\n1.Add Servers 2.Remove Servers 3.Add Processes 4.Remove Processes 5.Exit");

        // Add servers
        System.out.println("\nHow many more servers to add? 2");
        servers += 2;
        printLoad(servers, processes);

        System.out.println("\n1.Add Servers 2.Remove Servers 3.Add Processes 4.Remove Processes 5.Exit");

        // Remove processes
        System.out.println("\nHow many more Processes to remove? 2");
        processes -= 2;
        printLoad(servers, processes);

        System.out.println("\nProgram Ended");
    }
}
